244  static inline const kMIDIType getTypeFromStatusByte(const byte inStatus)
245  {
246  if ((inStatus < 0x80)
247  || (inStatus == 0xF4)
248  || (inStatus == 0xF5)
249  || (inStatus == 0xF9)
250  || (inStatus == 0xFD)) return InvalidType; // data bytes and undefined.
251  if (inStatus < 0xF0) return (kMIDIType)(inStatus & 0xF0); // Channel message, remove channel nibble.
252  else return (kMIDIType)inStatus;
253  }
